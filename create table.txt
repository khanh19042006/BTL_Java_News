CREATE TABLE news (
    id VARCHAR(255) PRIMARY KEY,
    link VARCHAR(512) NOT NULL,
    headline TEXT,
    category VARCHAR(255),
    short_description TEXT,
    authors TEXT,
    date VARCHAR(50)
);

--Thêm table category
CREATE TABLE category (
    id CHAR(36) PRIMARY KEY,
    code VARCHAR(50) NOT NULL UNIQUE,
    name VARCHAR(255) NOT NULL
);


--Thêm table role (admin, user, journalist)
CREATE TABLE role (
    id VARCHAR(36) PRIMARY KEY,
    code VARCHAR(50) NOT NULL UNIQUE,
    name VARCHAR(100) NOT NULL
);

INSERT IGNORE INTO role (id, code, name) VALUES
(UUID(), 'admin', 'Administrator'),
(UUID(), 'user', 'User'),
(UUID(), 'journalist', 'Journalist');

--Format lại các thuộc tính của news cho phù hợp
SET SQL_SAFE_UPDATES = 0;
UPDATE news n
JOIN category c
  ON LOWER(TRIM(n.category)) = LOWER(TRIM(c.name))
SET n.category = c.code;
SET SQL_SAFE_UPDATES = 1;

--Tạo table Token
CREATE TABLE token (
    id VARCHAR(36) NOT NULL,
    user_id VARCHAR(36) NOT NULL,
    otp_code VARCHAR(6) NOT NULL,
    iat BIGINT NOT NULL,
    exp BIGINT NOT NULL,
    PRIMARY KEY (id)
)

--Xây dựng lại table user
drop table users;
CREATE TABLE user (
    id VARCHAR(50) NOT NULL,
    username VARCHAR(100) NOT NULL,
    email VARCHAR(150) NOT NULL,
    password VARCHAR(255) NOT NULL,
    created_at VARCHAR(50) DEFAULT NULL,
    role VARCHAR(50) NOT NULL DEFAULT 'USER',
    isVerity TINYINT(1) DEFAULT '0',
    PRIMARY KEY (id),
    UNIQUE KEY username (username)
);
